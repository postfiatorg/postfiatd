57std::vector<std::string>
58NetworkValidators::getValidators(std::uint32_t networkId)
59{
60 switch (static_cast<NetworkType>(networkId))
61 {
62 case NetworkType::MAINNET:
63 return getMainnetValidators();
64 case NetworkType::TESTNET:
65 return getTestnetValidators();
66 case NetworkType::DEVNET:
67 return getDevnetValidators();
68 default:
69 // Default to mainnet validators for unknown network IDs
70 return getMainnetValidators();
71 }
72}
