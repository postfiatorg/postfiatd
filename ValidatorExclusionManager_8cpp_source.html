41std::optional<std::pair<std::optional<AccountID>, std::optional<AccountID>>>
42ValidatorExclusionManager::getExclusionChange(LedgerIndex ledgerSeq)
43{
44 std::lock_guard lock(mutex_);
45
46 // If not initialized yet, return nothing
47 if (!initialized_)
48 {
49 JLOG(j_.trace()) << "ValidatorExclusionManager: Not initialized yet";
50 return std::nullopt;
51 }
52
53 // If remote fetcher is configured, check its status first
54 if (remoteFetcher_)
55 {
56 // If initial fetch isn't complete or sources aren't accessible, no changes allowed
57 if (!remoteFetcher_->isInitialFetchComplete())
58 {
59 JLOG(j_.debug()) << "ValidatorExclusionManager: Remote fetcher not ready, no changes allowed";
60 return std::nullopt;
61 }
62
63 if (!remoteFetcher_->areAllSourcesAccessible())
64 {
65 JLOG(j_.debug()) << "ValidatorExclusionManager: Remote sources not accessible, no changes allowed";
66 return std::nullopt;
67 }
68
69 // Check if remote list has been modified
70 if (remoteFetcher_->hasBeenModified(true)) // Reset flag after checking
71 {
72 JLOG(j_.info()) << "ValidatorExclusionManager: Remote exclusion list modified, updating pending changes";
73
74 // Update ExclusionManager with reason information from remote fetcher
75 auto reasons = remoteFetcher_->getExclusionReasons();
76 if (!reasons.empty())
77 {
78 std::unordered_map<AccountID, ExclusionManager::ExclusionInfo> exclusionInfoMap;
79 for (auto const& [account, reasonPair] : reasons)
80 {
81 ExclusionManager::ExclusionInfo info;
82 info.reason = reasonPair.first;
83 info.dateAdded = reasonPair.second;
84 exclusionInfoMap[account] = info;
85 }
86 app_.getExclusionManager().updateExclusionReasons(exclusionInfoMap);
87 }
88
89 // Read current exclusions from the ledger
90 auto currentExclusions = getCurrentExclusionsFromLedger();
91
92 updatePendingChanges(currentExclusions);
93 }
94 }
95
96 // Check rate limiting - only allow changes every CHANGE_INTERVAL ledgers
97 if (ledgerSeq < lastChangeLedger_ + CHANGE_INTERVAL)
98 {
99 JLOG(j_.trace()) << "ValidatorExclusionManager: Rate limited, next change at ledger "
100 << (lastChangeLedger_ + CHANGE_INTERVAL);
101 return std::nullopt;
102 }
103
104 // Check if we have pending changes
105 if (pendingChanges_.empty())
106 {
107 JLOG(j_.trace()) << "ValidatorExclusionManager: No pending changes";
108 return std::nullopt;
109 }
110
111 // Get next change from queue and remove it
112 auto const [isAdd, account] = pendingChanges_.front();
113 pendingChanges_.pop();
114
115 // Update last change ledger
116 lastChangeLedger_ = ledgerSeq;
117
118 JLOG(j_.info()) << "ValidatorExclusionManager: Providing "
119 << (isAdd ? "add" : "remove")
120 << " for " << toBase58(account)
121 << " at ledger " << ledgerSeq
122 << ", " << pendingChanges_.size() << " changes remaining";
123
124 // Return the appropriate change
125 if (isAdd)
126 return std::make_pair(std::make_optional(account), std::nullopt);
127 else
128 return std::make_pair(std::nullopt, std::make_optional(account));
129}

131void
132ValidatorExclusionManager::initialize(
133 PublicKey const& validatorPubKey,
134 ReadView const& view)
135{
136 std::lock_guard lock(mutex_);
137
138 // Check if already initialized
139 if (initialized_)
140 {
141 return;
142 }
143
144 // Check if feature is enabled
145 if (!view.rules().enabled(featureAccountExclusion))
146 {
147 JLOG(j_.info()) << "ValidatorExclusionManager: AccountExclusion feature not enabled";
148 return;
149 }
150
151 // Get validator's master key from the signing key
152 // The validatorPubKey parameter is the signing key from app_.getValidationPublicKey()
153 // We need to get the master key to derive the correct account, matching what ValidatorVote does
154 auto masterKey = app_.validators().getTrustedKey(validatorPubKey);
155
156 // If not trusted, try listed key
157 if (!masterKey)
158 masterKey = app_.validators().getListedKey(validatorPubKey);
159
160 // Use master key if available, otherwise use signing key (same logic as RCLValidations)
161 auto const keyForAccount = masterKey.value_or(validatorPubKey);
162
163 // Get validator's account using the correct key
164 AccountID validatorAccount = calcAccountID(keyForAccount);
165
166 JLOG(j_.info()) << "ValidatorExclusionManager: Using "
167 << (masterKey ? "master key" : "signing key")
168 << " to derive account: " << toBase58(validatorAccount);
169
170 // Store validator account for future use
171 validatorAccount_ = validatorAccount;
172
173 // Get validator's current exclusion list from ledger
174 std::unordered_set<AccountID> currentExclusions;
175
176 auto const accountSLE = view.read(keylet::account(validatorAccount));
177 if (accountSLE && accountSLE->isFieldPresent(sfExclusionList))
178 {
179 auto const& exclusionList = accountSLE->getFieldArray(sfExclusionList);
180 for (auto const& entry : exclusionList)
181 {
182 if (entry.isFieldPresent(sfAccount))
183 {
184 currentExclusions.insert(entry.getAccountID(sfAccount));
185 }
186 }
187 }
188
189 JLOG(j_.info()) << "ValidatorExclusionManager: Initialized with "
190 << "current exclusions: " << currentExclusions.size()
191 << ", configured: " << configuredExclusions_.size();
192
193 // Update pending changes based on differences
194 updatePendingChanges(currentExclusions);
195
196 // Mark as initialized
197 initialized_ = true;
198}

200void
201ValidatorExclusionManager::updatePendingChanges(
202 std::unordered_set<AccountID> const& currentExclusions)
203{
204 // Note: This function should be called with mutex_ already locked
205 // Clear existing pending changes
206 while (!pendingChanges_.empty())
207 pendingChanges_.pop();
208
209 // If remote fetcher is configured, we must wait for it to be ready
210 if (remoteFetcher_)
211 {
212 // Check if initial fetch is complete and all sources are accessible
213 if (!remoteFetcher_->isInitialFetchComplete())
214 {
215 JLOG(j_.info()) << "ValidatorExclusionManager: Remote fetcher not ready yet, "
216 << "no exclusion changes will be made";
217 // Clear any pending changes and return without making any changes
218 while (!pendingChanges_.empty())
219 pendingChanges_.pop();
220 return;
221 }
222
223 if (!remoteFetcher_->areAllSourcesAccessible())
224 {
225 JLOG(j_.warn()) << "ValidatorExclusionManager: Not all remote sources accessible, "
226 << "no exclusion changes will be made";
227 // Clear any pending changes and return without making any changes
228 while (!pendingChanges_.empty())
229 pendingChanges_.pop();
230 return;
231 }
232 }
233
234 // Combine configured exclusions with remote exclusions
235 std::unordered_set<AccountID> combinedExclusions = configuredExclusions_;
236
237 if (remoteFetcher_)
238 {
239 // At this point we know the fetcher is ready and all sources are accessible
240 auto remoteExclusions = remoteFetcher_->getCombinedExclusions();
241 combinedExclusions.insert(remoteExclusions.begin(), remoteExclusions.end());
242
243 JLOG(j_.debug()) << "ValidatorExclusionManager: Added "
244 << remoteExclusions.size() << " remote exclusions";
245 }
246
247 // Find accounts to add (in combined config but not in ledger)
248 for (auto const& account : combinedExclusions)
249 {
250 if (currentExclusions.find(account) == currentExclusions.end())
251 {
252 pendingChanges_.push({true, account}); // true = add
253 JLOG(j_.debug()) << "ValidatorExclusionManager: Queued add for "
254 << toBase58(account);
255 }
256 }
257
258 // Find accounts to remove (in ledger but not in combined config)
259 for (auto const& account : currentExclusions)
260 {
261 if (combinedExclusions.find(account) == combinedExclusions.end())
262 {
263 pendingChanges_.push({false, account}); // false = remove
264 JLOG(j_.debug()) << "ValidatorExclusionManager: Queued remove for "
265 << toBase58(account);
266 }
267 }
268
269 JLOG(j_.info()) << "ValidatorExclusionManager: "
270 << pendingChanges_.size() << " pending changes queued";
271}

299std::unordered_set<AccountID>
300ValidatorExclusionManager::getCurrentExclusionsFromLedger() const
301{
302 std::unordered_set<AccountID> exclusions;
303
304 // Check if we have a validator account
305 if (!validatorAccount_)
306 {
307 JLOG(j_.warn()) << "ValidatorExclusionManager: No validator account set, cannot read from ledger";
308 return exclusions;
309 }
310
311 // Get the current ledger
312 auto ledger = app_.getLedgerMaster().getClosedLedger();
313 if (!ledger)
314 {
315 JLOG(j_.warn()) << "ValidatorExclusionManager: No closed ledger available";
316 return exclusions;
317 }
318
319 // Read the validator's account from ledger
320 auto const accountSLE = ledger->read(keylet::account(*validatorAccount_));
321 if (!accountSLE)
322 {
323 JLOG(j_.debug()) << "ValidatorExclusionManager: Validator account not found in ledger";
324 return exclusions;
325 }
326
327 // Read exclusion list if present
328 if (accountSLE->isFieldPresent(sfExclusionList))
329 {
330 auto const& exclusionList = accountSLE->getFieldArray(sfExclusionList);
331 for (auto const& entry : exclusionList)
332 {
333 if (entry.isFieldPresent(sfAccount))
334 {
335 exclusions.insert(entry.getAccountID(sfAccount));
336 }
337 }
338 }
339
340 JLOG(j_.debug()) << "ValidatorExclusionManager: Read " << exclusions.size()
341 << " exclusions from ledger";
342
343 return exclusions;
344}
