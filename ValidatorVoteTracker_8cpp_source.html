42void
43ValidatorVoteTracker::doVoting(
44 std::shared_ptr<ReadView const> const& lastClosedLedger,
45 std::vector<std::shared_ptr<STValidation>> const& parentValidations,
46 std::shared_ptr<SHAMap> const& initialPosition,
47 beast::Journal journal)
48{
49 // Check if the amendment is enabled
50 if (!isEnabled(lastClosedLedger->rules()))
51 {
52 JLOG(journal.trace()) << "ValidatorVoteTracker: Amendment not enabled";
53 return;
54 }
55
56 std::lock_guard lock(mutex_);
57
58 // Get the current ledger sequence
59 LedgerIndex currentSeq = lastClosedLedger->seq();
60
61 // Process votes for the last closed ledger (currentSeq)
62 // These votes will be included in the next ledger being built (currentSeq + 1)
63 LedgerIndex voteSeq = currentSeq;
64
65 auto it = votesByLedger_.find(voteSeq);
66 if (it == votesByLedger_.end() || it->second.empty())
67 {
68 JLOG(journal.trace()) << "ValidatorVoteTracker: No votes for ledger " << voteSeq;
69 return;
70 }
71
72 // Get the set of validators we've already processed for this ledger
73 auto& processed = processedValidators_[currentSeq];
74
75 // Generate ValidatorVote pseudo-transactions for each unique validator
76 std::set<PublicKey> uniqueValidators;
77 for (auto const& vote : it->second)
78 {
79 // Skip if we've already processed this validator for this ledger
80 if (processed.count(vote.validatorKey) > 0)
81 continue;
82
83 // Skip duplicate validators in this batch
84 if (!uniqueValidators.insert(vote.validatorKey).second)
85 continue;
86
87 // Create the ValidatorVote pseudo-transaction
88 STTx voteTx(
89 ttVALIDATOR_VOTE,
90 [this, &vote, seq = currentSeq + 1, &journal](auto& obj) {
91 obj.setAccountID(sfAccount, AccountID());
92 obj.setFieldU32(sfNetworkID, app_.config().NETWORK_ID);
93 obj.setFieldU32(sfLedgerSequence, vote.ledgerSeq);
94 obj.setFieldH256(sfLedgerHash, vote.ledgerHash);
95 obj.setFieldVL(sfValidatorPublicKey, vote.validatorKey.slice());
96 obj.setFieldH256(sfValidationHash, vote.validationHash);
97 if (vote.voteTime != NetClock::time_point{})
98 obj.setFieldU32(sfCloseTime, vote.voteTime.time_since_epoch().count());
99
100 // Add exclusion fields if present
101 if (vote.exclusionAdd) {
102 obj.setAccountID(sfExclusionAdd, *vote.exclusionAdd);
103 }
104 if (vote.exclusionRemove) {
105 obj.setAccountID(sfExclusionRemove, *vote.exclusionRemove);
106 }
107 });
108
109 Serializer s;
110 voteTx.add(s);
111
112 // Add to the transaction set
113 auto const txID = voteTx.getTransactionID();
114 auto const result = initialPosition->addItem(
115 SHAMapNodeType::tnTRANSACTION_NM,
116 make_shamapitem(txID, s.slice()));
117
118 // Mark this validator as processed for this ledger
119 processed.insert(vote.validatorKey);
120
121 JLOG(journal.info()) << "ValidatorVoteTracker: Generated vote transaction "
122 << txID
123 << " for validator " << toBase58(calcAccountID(vote.validatorKey))
124 << " in ledger " << currentSeq + 1
125 << " (addItem result: " << (result ? "success" : "failed") << ")";
126 }
127
128 // Clean up old data
129 cleanup(currentSeq);
130}

132void
133ValidatorVoteTracker::cleanup(LedgerIndex currentLedger)
134{
135 // Keep votes for the last 10 ledgers
136 const LedgerIndex keepAfter = (currentLedger > 10) ? currentLedger - 10 : 0;
137
138 // Clean up old votes
139 auto voteIt = votesByLedger_.begin();
140 while (voteIt != votesByLedger_.end())
141 {
142 if (voteIt->first < keepAfter)
143 voteIt = votesByLedger_.erase(voteIt);
144 else
145 ++voteIt;
146 }
147
148 // Clean up processed validators tracking
149 auto procIt = processedValidators_.begin();
150 while (procIt != processedValidators_.end())
151 {
152 if (procIt->first < keepAfter)
153 procIt = processedValidators_.erase(procIt);
154 else
155 ++procIt;
156 }
157}
