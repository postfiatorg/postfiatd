159void
160RemoteExclusionListFetcher::makeRequest(
161 Config::ValidatorExclusionSource const& source,
162 std::size_t sourceIdx)
163{
164 parsedURL pUrl;
165 if (!parseUrl(pUrl, source.url))
166 {
167 throw std::runtime_error("Invalid URL: " + source.url);
168 }
169
170 std::shared_ptr<detail::Work> sp;
171
172 auto onFetch = [this, sourceIdx](
173 error_code const& ec,
174 boost::asio::ip::tcp::endpoint const& endpoint,
175 detail::response_type&& res) {
176 // We don't use the endpoint, but it's required by the callback signature
177 onFetchComplete(ec, std::move(res), sourceIdx);
178 };
179
180 auto onFileFetch = [this, sourceIdx](
181 error_code const& ec,
182 std::string const& res) {
183 this->onTextFetch(ec, res, sourceIdx);
184 };
185
186 JLOG(j_.debug()) << "RemoteExclusionListFetcher: Starting request for " << source.url;
187
188 if (pUrl.scheme == "https")
189 {
190 if (pUrl.domain.empty())
191 throw std::runtime_error("https URI must contain a hostname");
192
193 if (!pUrl.port)
194 pUrl.port = 443;
195
196 sp = std::make_shared<detail::WorkSSL>(
197 pUrl.domain,
198 pUrl.path.empty() ? "/" : pUrl.path,
199 std::to_string(*pUrl.port),
200 app_.getIOService(),
201 j_,
202 app_.config(),
203 boost::asio::ip::tcp::endpoint{}, // No cached endpoint
204 false, // Not using cached endpoint
205 onFetch);
206 }
207 else if (pUrl.scheme == "http")
208 {
209 if (pUrl.domain.empty())
210 throw std::runtime_error("http URI must contain a hostname");
211
212 if (!pUrl.port)
213 pUrl.port = 80;
214
215 sp = std::make_shared<detail::WorkPlain>(
216 pUrl.domain,
217 pUrl.path.empty() ? "/" : pUrl.path,
218 std::to_string(*pUrl.port),
219 app_.getIOService(),
220 boost::asio::ip::tcp::endpoint{}, // No cached endpoint
221 false, // Not using cached endpoint
222 onFetch);
223 }
224 else if (pUrl.scheme == "file")
225 {
226 if (!pUrl.domain.empty())
227 throw std::runtime_error("file URI cannot contain a hostname");
228
229#if BOOST_OS_WINDOWS
230 // Paths on Windows need the leading / removed
231 if (pUrl.path[0] == '/')
232 pUrl.path = pUrl.path.substr(1);
233#endif
234
235 if (pUrl.path.empty())
236 throw std::runtime_error("file URI must contain a path");
237
238 sp = std::make_shared<detail::WorkFile>(
239 pUrl.path,
240 app_.getIOService(),
241 onFileFetch);
242 }
243 else
244 {
245 throw std::runtime_error("Unsupported scheme: " + pUrl.scheme);
246 }
247
248 work_ = sp;
249 sp->run();
250}

253RemoteExclusionListFetcher::onFetchComplete(

255 detail::response_type&& res,
256 std::size_t sourceIdx)
257{
258 auto const& source = config_.VALIDATOR_EXCLUSIONS_SOURCES[sourceIdx];
259
260 if (ec)

264
265 fetchResults_[sourceIdx].success = false;
266 fetchResults_[sourceIdx].errorMessage = ec.message();

268 else
269 {
270 using namespace boost::beast::http;
271

276 }
277 else
278 {
279 JLOG(j_.warn()) << "RemoteExclusionListFetcher: Bad HTTP status from "
280 << source.url << ": " << res.result_int();
281
282 fetchResults_[sourceIdx].success = false;
283 fetchResults_[sourceIdx].errorMessage = "HTTP " + std::to_string(res.result_int());
284 }
285 }
286
287 // Check if all fetches are complete
288 checkAllFetchesComplete();

291void
292RemoteExclusionListFetcher::onTextFetch(
293 error_code const& ec,
294 std::string const& res,
295 std::size_t sourceIdx)
296{
297 auto const& source = config_.VALIDATOR_EXCLUSIONS_SOURCES[sourceIdx];
298
299 if (ec)
300 {
301 JLOG(j_.warn()) << "RemoteExclusionListFetcher: Error reading from "
302 << source.url << ": " << ec.message();
303
304 fetchResults_[sourceIdx].success = false;
305 fetchResults_[sourceIdx].errorMessage = ec.message();
306 }
307 else
308 {
309 // Parse and verify the response
310 auto list = parseExclusionList(res);
311 if (list && verifySignature(*list, source.pubkey, res))
312 {
313 list->verified = true;
314 fetchResults_[sourceIdx].success = true;
315 fetchResults_[sourceIdx].list = list;
316
317 JLOG(j_.info()) << "RemoteExclusionListFetcher: Successfully fetched and verified from "
318 << source.url;
319 }
320 else
321 {
322 JLOG(j_.error()) << "RemoteExclusionListFetcher: Failed to parse or verify from "
323 << source.url;
324
325 fetchResults_[sourceIdx].success = false;
326 fetchResults_[sourceIdx].errorMessage = "Parse/verify failed";
327 }
328 }
329
330 checkAllFetchesComplete();

350void
351RemoteExclusionListFetcher::onAllFetchesComplete()
352{
353 JLOG(j_.debug()) << "RemoteExclusionListFetcher: All fetches complete";
354
355 std::unordered_map<std::string, ExclusionList> newLists;
356 bool allSuccessful = true;
357 size_t successCount = 0;
358
359 for (std::size_t i = 0; i < fetchResults_.size(); ++i)
360 {
361 auto const& source = config_.VALIDATOR_EXCLUSIONS_SOURCES[i];
362 auto const& result = fetchResults_[i];
363
364 if (result.success && result.list)
365 {
366 newLists[source.url] = *result.list;
367 successCount++;
368 }
369 else
370 {
371 allSuccessful = false;
372 }
373 }
374
375 // Update cached lists and combined exclusions
376 {
377 std::lock_guard lock(mutex_);
378
379 // Only update if ALL sources were successfully fetched
380 // or if this is not the initial fetch and we had some successes
381 if (allSuccessful)
382 {
383 cachedLists_ = std::move(newLists);
384 allSourcesAccessible_ = true;
385 lastSuccessfulFetchTime_ = std::chrono::steady_clock::now();
386 updateCombinedExclusions();
387
388 JLOG(j_.info()) << "RemoteExclusionListFetcher: All sources accessible, updating exclusions";
389 }
390 else if (initialFetchComplete_ && successCount > 0)
391 {
392 // After initial fetch, we can continue with partial updates
393 // but we mark that not all sources are accessible
394 JLOG(j_.warn()) << "RemoteExclusionListFetcher: Only " << successCount << "/"
395 << config_.VALIDATOR_EXCLUSIONS_SOURCES.size()
396 << " sources accessible, keeping existing exclusions";
397 allSourcesAccessible_ = false;
398 }
399 else if (!initialFetchComplete_)
400 {
401 // On initial fetch, if we can't reach all sources, don't use any exclusions
402 JLOG(j_.error()) << "RemoteExclusionListFetcher: Initial fetch failed - not all sources accessible. "
403 << "No remote exclusions will be used.";
404 cachedLists_.clear();
405 combinedExclusions_.clear();
406 allSourcesAccessible_ = false;
407 }
408
409 initialFetchComplete_ = true;
410 }
411
412 fetching_ = false;
413}

415std::optional<RemoteExclusionListFetcher::ExclusionList>
416RemoteExclusionListFetcher::parseExclusionList(std::string const& content)
417{
418 try
419 {
420 Json::Value json;
421 Json::Reader reader;
422
423 if (!reader.parse(content, json))
424 {
425 JLOG(j_.error()) << "RemoteExclusionListFetcher: Failed to parse JSON";
426 return std::nullopt;
427 }
428
429 ExclusionList list;
430
431 // Parse basic fields
432 if (json.isMember("version"))
433 list.version = json["version"].asString();
434
435 if (json.isMember("timestamp"))
436 list.timestamp = json["timestamp"].asString();
437
438 if (json.isMember("issuer_address"))
439 {
440 auto issuer = parseBase58<AccountID>(json["issuer_address"].asString());
441 if (!issuer)
442 {
443 JLOG(j_.error()) << "RemoteExclusionListFetcher: Invalid issuer address";
444 return std::nullopt;
445 }
446 list.issuerAddress = *issuer;
447 }
448
449 // Parse exclusions entries
450 if (json.isMember("exclusions") && json["exclusions"].isArray())
451 {
452 for (auto const& entry : json["exclusions"])
453 {
454 ExclusionEntry exclusion;
455
456 if (entry.isMember("address"))
457 {
458 auto addr = parseBase58<AccountID>(entry["address"].asString());
459 if (!addr)
460 {
461 JLOG(j_.warn()) << "RemoteExclusionListFetcher: Skipping invalid address in exclusions";
462 continue;
463 }
464 exclusion.address = *addr;
465 }
466
467 if (entry.isMember("reason"))
468 exclusion.reason = entry["reason"].asString();
469
470 if (entry.isMember("date_added"))
471 exclusion.dateAdded = entry["date_added"].asString();
472
473 list.blacklist.push_back(std::move(exclusion));
474 }
475 }
476
477 return list;
478 }
479 catch (std::exception const& e)
480 {
481 JLOG(j_.error()) << "RemoteExclusionListFetcher: Parse error: " << e.what();
482 }
483
484 return std::nullopt;

487bool
488RemoteExclusionListFetcher::verifySignature(
489 ExclusionList const& list,
490 std::string const& pubkeyStr,
491 std::string const& rawContent)
492{
493 try
494 {
495 Json::Value json;
496 Json::Reader reader;
497
498 if (!reader.parse(rawContent, json))
499 return false;
500
501 if (!json.isMember("signature"))
502 return false;
503
504 auto const& sig = json["signature"];
505
506 std::string algorithm = sig.isMember("algorithm") ? sig["algorithm"].asString() : "";
507 std::string sigPubKey = sig.isMember("public_key") ? sig["public_key"].asString() : "";
508 std::string signature = sig.isMember("signature") ? sig["signature"].asString() : "";
509
510 // Verify the public key matches what's configured
511 if (sigPubKey != pubkeyStr)
512 {
513 JLOG(j_.error()) << "RemoteExclusionListFetcher: Public key mismatch";
514 return false;
515 }
516
517 // Create a canonical message to verify by concatenating sorted addresses
518 // This avoids JSON serialization issues
519 std::vector<std::string> addresses;
520 addresses.reserve(list.blacklist.size());
521
522 for (auto const& entry : list.blacklist)
523 {
524 addresses.push_back(toBase58(entry.address));
525 }
526
527 // Sort addresses to ensure consistent ordering
528 std::sort(addresses.begin(), addresses.end());
529
530 // Create message as concatenated sorted addresses
531 std::string message;
532 for (auto const& addr : addresses)
533 {
534 message += addr;
535 message+="\n";
536 }
537
538 // Add version and timestamp for additional verification
539 message = "v1:" + list.version + ":" + list.timestamp + ":" +
540 toBase58(list.issuerAddress) + ":" + message;
541
542 // Parse the public key
543 auto pubkey = parseBase58<PublicKey>(TokenType::NodePublic, pubkeyStr);
544 if (!pubkey)
545 {
546 JLOG(j_.error()) << "RemoteExclusionListFetcher: Invalid public key format";
547 return false;
548 }
549
550 // Verify based on algorithm
551 if (algorithm == "ed25519")
552 {
553 return verifyEd25519Signature(message, signature, *pubkey);
554 }
555 else if (algorithm == "secp256k1")
556 {
557 return verifySecp256k1Signature(message, signature, *pubkey);
558 }
559 else
560 {
561 JLOG(j_.error()) << "RemoteExclusionListFetcher: Unsupported algorithm: " << algorithm;
562 return false;
563 }
564 }
565 catch (std::exception const& e)
566 {
567 JLOG(j_.error()) << "RemoteExclusionListFetcher: Signature verification error: " << e.what();
568 }
569
570 return false;
571}

575bool
576RemoteExclusionListFetcher::verifyEd25519Signature(
577 std::string const& message,
578 std::string const& signatureHex,
579 PublicKey const& pubkey)
580{
581 try
582 {
583 // Decode hex signature
584 auto sigDecoded = strUnHex(signatureHex);
585 if (!sigDecoded || sigDecoded->size() != 64)
586 {
587 JLOG(j_.error()) << "RemoteExclusionListFetcher: Invalid Ed25519 signature size";
588 return false;
589 }
590
591 // Verify the signature using verify
592 return verify(pubkey, makeSlice(message), makeSlice(*sigDecoded), true);
593 }
594 catch (std::exception const& e)
595 {
596 JLOG(j_.error()) << "RemoteExclusionListFetcher: Ed25519 verification error: " << e.what();
597 }
598
599 return false;

602bool
603RemoteExclusionListFetcher::verifySecp256k1Signature(
604 std::string const& message,
605 std::string const& signatureHex,
606 PublicKey const& pubkey)
607{
608 try
609 {
610 // Decode hex signature
611 auto sigDecoded = strUnHex(signatureHex);
612 if (!sigDecoded)
613 {
614 JLOG(j_.error()) << "RemoteExclusionListFetcher: Failed to decode secp256k1 signature";
615 return false;
616 }
617
618 // Hash the message
619 auto const messageHash = sha512Half(makeSlice(message));
620
621 // Verify the signature using verifyDigest
622 return verifyDigest(pubkey, messageHash, makeSlice(*sigDecoded), false);
623 }
624 catch (std::exception const& e)
625 {
626 JLOG(j_.error()) << "RemoteExclusionListFetcher: Secp256k1 verification error: " << e.what();
627 }
628
629 return false;
