34Json::Value
35doExclusionInfo(RPC::JsonContext& context)
36{
37 auto const& params = context.params;
38 Json::Value result;
39
40 // Get the current ledger
41 std::shared_ptr<ReadView const> ledger;
42 auto res = RPC::lookupLedger(ledger, context);
43 if (!ledger)
44 return res;
45
46 JLOG(context.j.info()) << "ExclusionInfo: Checking ledger " << ledger->seq();
47
48 // Check if AccountExclusion feature is enabled
49 if (!ledger->rules().enabled(featureAccountExclusion))
50 {
51 RPC::inject_error(rpcNOT_ENABLED, result);
52 result[jss::error_message] = "AccountExclusion feature is not enabled";
53 return result;
54 }
55
56 auto& app = context.app;
57 auto& exclusionManager = app.getExclusionManager();
58 auto const& validators = app.validators();
59
60 // Check if a specific validator was requested
61 if (params.isMember(jss::validator))
62 {
63 AccountID validatorAccount;
64
65 auto const validator = params[jss::validator].asString();
66 if (auto const account = parseBase58<AccountID>(validator))
67 {
68 validatorAccount = *account;
69 }
70 else
71 {
72 RPC::inject_error(rpcACT_MALFORMED, result);
73 return result;
74 }
75
76 // Get the validator's account
77 auto const accountSLE = ledger->read(keylet::account(validatorAccount));
78 if (!accountSLE)
79 {
80 RPC::inject_error(rpcACT_NOT_FOUND, result);
81 return result;
82 }
83
84 result[jss::validator] = toBase58(validatorAccount);
85 Json::Value& exclusionList = (result[jss::exclusion_list] = Json::arrayValue);
86
87 // Get the exclusion list for this validator
88 if (accountSLE->isFieldPresent(sfExclusionList))
89 {
90 STArray const& list = accountSLE->getFieldArray(sfExclusionList);
91 for (auto const& entry : list)
92 {
93 if (entry.isFieldPresent(sfAccount))
94 {
95 auto const excludedAccount = entry.getAccountID(sfAccount);
96 Json::Value exclusionEntry;
97 exclusionEntry["address"] = toBase58(excludedAccount);
98
99 // Add reason information if available from ExclusionManager
100 auto exclusionInfo = exclusionManager.getExclusionInfo(excludedAccount);
101 if (exclusionInfo)
102 {
103 if (!exclusionInfo->reason.empty())
104 exclusionEntry["reason"] = exclusionInfo->reason;
105 if (!exclusionInfo->dateAdded.empty())
106 exclusionEntry["date_added"] = exclusionInfo->dateAdded;
107 }
108
109 exclusionList.append(exclusionEntry);
110 }
111 }
112 }
113
114 result[jss::exclusion_count] = exclusionList.size();
115 }
116 else
117 {
118 // Return information for all validators
119 Json::Value& validatorsList = (result[jss::validators] = Json::objectValue);
120 Json::Value& excludedAccounts = (result[jss::excluded_accounts] = Json::objectValue);
121
122 std::unordered_map<AccountID, std::size_t> exclusionCounts;
123 std::size_t totalValidators = 0;
124
125 // Get list of trusted validators from the UNL
126 auto trustedKeys = validators.getTrustedMasterKeys();
127
128 for (auto const& pubKey : trustedKeys)
129 {
130 AccountID validatorAccount = calcAccountID(pubKey);
131 totalValidators++;
132
133 JLOG(context.j.info()) << "ExclusionInfo: Looking for validator with pubKey: "
134 << toBase58(TokenType::NodePublic, pubKey)
135 << " -> Account: " << toBase58(validatorAccount);
136
137 // Get the validator's account
138 auto const accountSLE = ledger->read(keylet::account(validatorAccount));
139 if (!accountSLE)
140 {
141 JLOG(context.j.info()) << "ExclusionInfo: Account not found for "
142 << toBase58(validatorAccount);
143 continue;
144 }
145 else
146 {
147 JLOG(context.j.info()) << "ExclusionInfo: Found account for "
148 << toBase58(validatorAccount);
149 }
150
151 Json::Value& validatorInfo = validatorsList[toBase58(validatorAccount)];
152 Json::Value& exclusionList = (validatorInfo[jss::exclusion_list] = Json::arrayValue);
153
154 // Get the exclusion list for this validator
155 if (accountSLE->isFieldPresent(sfExclusionList))
156 {
157 STArray const& list = accountSLE->getFieldArray(sfExclusionList);
158 for (auto const& entry : list)
159 {
160 if (entry.isFieldPresent(sfAccount))
161 {
162 auto const excludedAccount = entry.getAccountID(sfAccount);
163 exclusionList.append(toBase58(excludedAccount));
164 exclusionCounts[excludedAccount]++;
165 }
166 }
167 }
168
169 validatorInfo[jss::exclusion_count] = exclusionList.size();
170 }
171
172 // Calculate consensus threshold
173 ConsensusParms consensusParms{};
174 std::size_t const threshold =
175 (totalValidators * consensusParms.minCONSENSUS_PCT + 99) / 100;
176
177 result[jss::total_validators] = static_cast<Json::UInt>(totalValidators);
178 result[jss::consensus_threshold] = static_cast<Json::UInt>(threshold);
179 result[jss::consensus_percentage] = static_cast<Json::UInt>(consensusParms.minCONSENSUS_PCT);
180
181 // For each excluded account, show how many validators exclude it
182 for (auto const& [account, count] : exclusionCounts)
183 {
184 Json::Value& accountInfo = excludedAccounts[toBase58(account)];
185 accountInfo[jss::exclusion_count] = static_cast<Json::UInt>(count);
186 accountInfo["percentage"] =
187 static_cast<Json::UInt>((count * 100) / totalValidators);
188 accountInfo["meets_threshold"] = (count >= threshold);
189
190 // Add reason information if available from ExclusionManager
191 auto exclusionInfo = exclusionManager.getExclusionInfo(account);
192 if (exclusionInfo)
193 {
194 if (!exclusionInfo->reason.empty())
195 accountInfo["reason"] = exclusionInfo->reason;
196 if (!exclusionInfo->dateAdded.empty())
197 accountInfo["date_added"] = exclusionInfo->dateAdded;
198 }
199 }
200
201 // Get stats from ExclusionManager
202 auto stats = exclusionManager.getStats();
203 Json::Value& managerStats = result["exclusion_manager_stats"];
204 managerStats["total_validators_cached"] =
205 static_cast<Json::UInt>(stats.totalValidators);
206 managerStats["consensus_excluded_count"] =
207 static_cast<Json::UInt>(stats.totalExcludedAddresses);
208 managerStats["unique_exclusions"] =
209 static_cast<Json::UInt>(stats.totalUniqueExclusions);
210 }
211
212 return result;
213}
