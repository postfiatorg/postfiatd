58void
59ExclusionManager::updateValidatorExclusions(
60 AccountID const& validator,
61 std::unordered_set<AccountID> const& exclusions)
62{
63 std::lock_guard lock(mutable_);
64
65 // Get the old exclusion list for this validator (if any)
66 auto oldExclusionsIt = validatorExclusions_.find(validator);
67 std::unordered_set<AccountID> oldExclusions;
68
69 if (oldExclusionsIt != validatorExclusions_.end())
70 {
71 oldExclusions = std::move(oldExclusionsIt->second);
72 }
73
74 // Update to new exclusion list
75 if (exclusions.empty())
76 {
77 validatorExclusions_.erase(validator);
78 }
79 else
80 {
81 validatorExclusions_[validator] = exclusions;
82 }
83
84 // Update the total validator count to match the actual map size
85 totalValidators_ = validatorExclusions_.size();
86
87 // Update exclusion counts
88 // First, decrement counts for addresses that were in old list
89 for (auto const& account : oldExclusions)
90 {
91 auto it = exclusionCounts_.find(account);
92 if (it != exclusionCounts_.end())
93 {
94 if (it->second > 1)
95 {
96 it->second--;
97 }
98 else
99 {
100 exclusionCounts_.erase(it);
101 }
102 }
103 }
104
105 // Then, increment counts for addresses in new list
106 for (auto const& account : exclusions)
107 {
108 exclusionCounts_[account]++;
109 }
110
111 // Recalculate which addresses meet consensus threshold
112 recalculateConsensusExclusions();
113
114 JLOG(j_.debug()) << "Updated exclusions for validator " << toBase58(validator)
115 << " - " << exclusions.size() << " addresses excluded";
116}

118void
119ExclusionManager::removeValidator(AccountID const& validator)
120{
121 std::lock_guard lock(mutable_);
122
123 auto it = validatorExclusions_.find(validator);
124 if (it == validatorExclusions_.end())
125 return;
126
127 // Decrement counts for all addresses this validator was excluding
128 for (auto const& account : it->second)
129 {
130 auto countIt = exclusionCounts_.find(account);
131 if (countIt != exclusionCounts_.end())
132 {
133 if (countIt->second > 1)
134 {
135 countIt->second--;
136 }
137 else
138 {
139 exclusionCounts_.erase(countIt);
140 }
141 }
142 }
143
144 validatorExclusions_.erase(it);
145
146 // Update the total validator count to match the actual map size
147 totalValidators_ = validatorExclusions_.size();
148
149 recalculateConsensusExclusions();
150
151 JLOG(j_.debug()) << "Removed validator " << toBase58(validator)
152 << " from exclusion tracking";
153}

155void
156ExclusionManager::rebuildCache(ReadView const& view)
157{
158 if (!view.rules().enabled(featureAccountExclusion))
159 {
160 JLOG(j_.debug()) << "AccountExclusion feature not enabled, skipping cache rebuild";
161 return;
162 }
163
164 std::lock_guard lock(mutable_);
165
166 // Clear existing data
167 validatorExclusions_.clear();
168 exclusionCounts_.clear();
169 consensusExcluded_.clear();
170 totalValidators_ = 0;
171
172 JLOG(j_.info()) << "Rebuilding exclusion cache from ledger";
173
174 std::size_t validatorsProcessed = 0;
175 std::size_t totalExclusions = 0;
176
177 // Get list of trusted validators from the app's validator list
178 // These are the validators whose vote counts for consensus
179 auto const& validators = app_.validators();
180 auto trustedKeys = validators.getTrustedMasterKeys();
181
182 // For each trusted UNL validator, check if they have an exclusion list in their account
183 for (auto const& pubKey : trustedKeys)
184 {
185 // Convert public key to account ID
186 AccountID validatorAccount = calcAccountID(pubKey);
187
188 // Get the validator's account to check their exclusion list
189 auto const accountSLE = view.read(keylet::account(validatorAccount));
190 if (!accountSLE)
191 {
192 // This validator doesn't have an account yet, skip
193 JLOG(j_.debug()) << "UNL validator has no account: "
194 << toBase58(validatorAccount);
195 continue;
196 }
197
198 // This is a UNL validator, count them
199 totalValidators_++;
200 validatorsProcessed++;
201
202 // Check if this validator has an exclusion list
203 if (accountSLE->isFieldPresent(sfExclusionList))
204 {
205 std::unordered_set<AccountID> exclusions;
206 STArray const& exclusionList = accountSLE->getFieldArray(sfExclusionList);
207
208 for (auto const& entry : exclusionList)
209 {
210 if (entry.isFieldPresent(sfAccount))
211 {
212 auto const excludedAccount = entry.getAccountID(sfAccount);
213 exclusions.insert(excludedAccount);
214 exclusionCounts_[excludedAccount]++;
215 totalExclusions++;
216 }
217 }
218
219 if (!exclusions.empty())
220 {
221 validatorExclusions_[validatorAccount] = std::move(exclusions);
222 }
223 }
224 }
225
226 // Update the total validator count to match the actual map size
227 totalValidators_ = validatorExclusions_.size();
228
229 // Calculate which addresses meet consensus threshold
230 recalculateConsensusExclusions();
231
232 // Mark as initialized
233 initialized_ = true;
234
235 JLOG(j_.info()) << "Exclusion cache rebuilt: "
236 << validatorsProcessed << " validators, "
237 << totalExclusions << " total exclusions, "
238 << consensusExcluded_.size() << " addresses meet consensus threshold";
239}

241void
242ExclusionManager::recalculateConsensusExclusions()
243{
244 consensusExcluded_.clear();
245
246 if (totalValidators_ == 0)
247 return;
248
249 std::size_t const threshold = getConsensusThreshold();
250
251 for (auto const& [account, count] : exclusionCounts_)
252 {
253 if (count >= threshold)
254 {
255 consensusExcluded_.insert(account);
256 }
257 }
258
259 JLOG(j_.trace()) << "Consensus exclusions recalculated: "
260 << consensusExcluded_.size() << " addresses excluded "
261 << "(threshold: " << threshold << "/" << totalValidators_ << ")";
262}

264std::size_t
265ExclusionManager::getConsensusThreshold() const
266{
267 if (totalValidators_ == 0)
268 return 1;
269
270 // Use minCONSENSUS_PCT (67%) to determine threshold
271 ConsensusParms consensusParms{};
272 std::size_t const threshold =
273 (totalValidators_ * consensusParms.minCONSENSUS_PCT + 99) / 100;
274
275 return std::max(std::size_t(1), threshold);
276}

279ExclusionManager::getStats() const
280{
281 std::lock_guard lock(mutable_);
282
283 Stats stats;
284 stats.totalValidators = totalValidators_;
285 stats.totalExcludedAddresses = consensusExcluded_.size();
286
287 // Count total unique exclusions across all validators
288 std::unordered_set<AccountID> allExclusions;
289 for (auto const& [validator, exclusions] : validatorExclusions_)
290 {
291 allExclusions.insert(exclusions.begin(), exclusions.end());
292 }
293 stats.totalUniqueExclusions = allExclusions.size();
294
295 return stats;

298std::optional<ExclusionManager::ExclusionInfo>
299ExclusionManager::getExclusionInfo(AccountID const& account) const
300{
301 std::lock_guard lock(mutable_);
302
303 // Check if remote fetcher has updates and apply them
304 if (remoteFetcher_ && remoteFetcher_->hasBeenModified(false))
305 {
306 // Get updated reasons from remote fetcher
307 auto reasons = remoteFetcher_->getExclusionReasons();
308 if (!reasons.empty())
309 {
310 // Update the exclusion info map
311 // Note: We need to cast away const here since we're updating cache
312 // This is safe because the method is logically const (query operation)
313 auto* mutableThis = const_cast<ExclusionManager*>(this);
314 for (auto const& [acct, reasonPair] : reasons)
315 {
316 ExclusionInfo info;
317 info.reason = reasonPair.first;
318 info.dateAdded = reasonPair.second;
319 mutableThis->exclusionInfoMap_[acct] = info;
320 }
321
322 JLOG(j_.debug()) << "ExclusionManager: Updated reasons from remote fetcher for "
323 << reasons.size() << " addresses";
324 }
325 }
326
327 auto it = exclusionInfoMap_.find(account);
328 if (it != exclusionInfoMap_.end())
329 {
330 // Update vote count from current exclusion counts
331 ExclusionInfo info = it->second;
332 auto countIt = exclusionCounts_.find(account);
333 if (countIt != exclusionCounts_.end())
334 {
335 info.voteCount = countIt->second;
336 }
337 return info;
338 }
339
340 // If no info stored but account is in exclusion counts, return basic info
341 auto countIt = exclusionCounts_.find(account);
342 if (countIt != exclusionCounts_.end())
343 {
344 ExclusionInfo info;
345 info.voteCount = countIt->second;
346 return info;
347 }
348
349 return std::nullopt;
350}
